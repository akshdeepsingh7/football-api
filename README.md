## Using `standings.json`

The `standings.json` file contains all the standings data in a structured JSON format, making it easy to consume programmatically. Below are some common use cases and examples.

### JSON Structure

```json
{
  "generated_at": "2026-05-05 10:04:19 UTC",
  "generated_at_iso": "2026-05-05T10:04:19.085044+00:00",
  "source": "Public Football Data",
  "author": {
    "name": "Akshdeep Singh",
    "url": "https://github.com/akshdeepsingh7",
    "github": "https://github.com/akshdeepsingh7/football-standings"
  },
  "leagues": {
    "English Premier League": {
      "slug": "eng.1",
      "country": "England",
      "confederation": "UEFA",
      "founded": 1992,
      "description": "...",
      "team_count": 20,
      "teams": [
        {
          "rank": 1,
          "group": "...",
          "team": "Arsenal",
          "abbreviation": "ARS",
          "played": 35,
          "wins": 23,
          "draws": 7,
          "losses": 5,
          "goals_for": 67,
          "goals_against": 26,
          "goal_difference": 41,
          "points": 76,
          "record": "23-7-5"
        }
      ]
    }
  }
}
```

### Example: Python

```python
import json

with open("standings.json") as f:
    data = json.load(f)

# Get all EPL teams
epl = data["leagues"]["English Premier League"]["teams"]
for t in epl[:5]:
    print(f"#{t['rank']} {t['team']} - {t['points']} pts")

# Find the top scorer (most goals for) across all leagues
best_attack = None
for league_name, league in data["leagues"].items():
    for t in league["teams"]:
        if best_attack is None or t["goals_for"] > best_attack["goals_for"]:
            best_attack = t
            best_attack["league"] = league_name
print(f"Best attack: {best_attack['team']} ({best_attack['league']}) - {best_attack['goals_for']} GF")
```

### Example: JavaScript / Node.js

```javascript
const data = require("./standings.json");

// Get La Liga table
const laliga = data.leagues["Spanish La Liga"].teams;
laliga.forEach(t => {
  console.log(`#${t.rank} ${t.team} - ${t.points} pts (GD: ${t.goal_difference})`);
});
```

### Example: cURL

```bash
# Pretty-print the JSON
cat standings.json | python3 -m json.tool

# Get just the league names
cat standings.json | python3 -c "import json,sys; d=json.load(sys.stdin); [print(k) for k in d['leagues']]"
```
